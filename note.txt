Document 보는 방법 
 사용하고자 하는 모듈을 로딩해서 구현해야 함으로. 
 nodejs 홈페이지 API 문서 

 모듈의 구성 - 클래스/이벤트/모듈함수

 모듈 로딩 : require('모듈이름')
 //readline : https://nodejs.org/dist/latest-v14.x/docs/api/readline.html
 var readline = require('readline');
 기본 모듈 : node.js 라이브러리 디렉토리 
 확장 모듈 : 같은폴더 또는 [상위] node_modules 폴더 

 var rl = readline.createInterface({  // 객채 생성할 때, 파라메타가 들어가는 것들은 아래와 같이 키:값 으로 대입한다. 문서에 어떤 파라메타가 있는지 확인가능함.
     input: process.stdin,
     output: process.stdout
 })


 이벤트 
 .on([이벤트 이름],[리스너 함수])
 rl.on('line',function(cmd){
     console.log('event : '+cmd);
 })
